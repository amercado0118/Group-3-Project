start

	display group members
	

	display "Group 3 Pizza Ordering System"
	display "Pizza Prices"
	display "Small Pizza: $15"
	display "Medium Pizza: $20"
	display "Large Pizza: $25"
	display "Pepperoni Topping: $2 (Small), $3 (Medium/Large)"
	display "Extra Cheese: $1 (All sizes)"
	

	input pizzaSize
	convert size letter to uppercase

	while pizzaSize is not S, M, L
		display "Invalid size. Enter S, M, L"
		input pizzaSize
		convert pizzaSize to uppercase


	input pepperoniChoice
	convert pepperoniChoice to uppercase

	while pepperoniChoice is not Y/N
		display "Invalid input. Enter Y or N"
		input pepperoniChoice
		convert pepperoniChoice to uppercase


	input cheeseChoice
	convert cheeseChoice to uppercase


	while cheeseChoice is not Y/N
		display "Invalid input. Enter Y or N"
		input cheeseChoice
		convert cheeseChoice to uppercase

	
	look up basePrice using pizzaSize
	set total = basePrice

	
	set pepperoniPrice = 0
	if pepperoniChoice equals Y
		look up pepperoniPrice using pizzaSize
		set total = total + pepperoniPrice


	set cheesePrice = 0
	if cheeseChoice equals Y
		look up cheesePrice using pizzaSize
		set total = total + cheesePrice


	display "Order Summary"
	display pizzaSize
	display pepperoniChoice
	display cheeseChoice
	display total

	display "thanks"

stop
